# mac iterm2 key

```bash
# 分屏

# 垂直
command + d
# 水平
command + shift + d  
# 切换屏幕
command + option + 方向键
command + [|]

# 查看历史命令
command + ;

# 查看剪贴板历史
command + shift + h

# 清除当前行
ctrl + u
# 前进／后退
ctrl + f/b
# 上一条命令
ctrl + p

# 删除当前光标字符
ctrl + d
# 删除当前之前光标的字符
ctrl + h
# 删除光标之前的单词
ctrl + w
# 删除光标到末尾
ctrl + k
# 交换光标处文本
ctrl + t

```
